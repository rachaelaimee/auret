// Firebase Firestore Security Rules
// Copy these rules to your Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow read: if request.auth != null; // Allow authenticated users to read other profiles
    }
    
    // Shops collection
    match /shops/{shopId} {
      allow read: if true; // Public read for shop discovery
      allow write: if request.auth != null && 
        (resource == null || resource.data.ownerId == request.auth.uid);
    }
    
    // Products collection
    match /products/{productId} {
      allow read: if true; // Public read for product browsing
      allow create: if request.auth != null && 
        exists(/databases/$(database)/documents/shops/$(resource.data.shopId)) &&
        get(/databases/$(database)/documents/shops/$(resource.data.shopId)).data.ownerId == request.auth.uid;
      allow update, delete: if request.auth != null && 
        exists(/databases/$(database)/documents/shops/$(resource.data.shopId)) &&
        get(/databases/$(database)/documents/shops/$(resource.data.shopId)).data.ownerId == request.auth.uid;
    }
    
    // Orders collection
    match /orders/{orderId} {
      allow read, write: if request.auth != null && 
        (resource.data.buyerId == request.auth.uid || 
         resource.data.sellerId == request.auth.uid);
    }
    
    // Tutorials collection
    match /tutorials/{tutorialId} {
      allow read: if resource == null || resource.data.status == 'published' || 
        (request.auth != null && resource.data.authorId == request.auth.uid);
      allow create: if request.auth != null && 
        request.auth.uid == resource.data.authorId;
      allow update, delete: if request.auth != null && 
        resource.data.authorId == request.auth.uid;
    }
    
    // Tutorial comments collection
    match /tutorials/{tutorialId}/comments/{commentId} {
      allow read: if true; // Public read for comments
      allow create: if request.auth != null && 
        request.auth.uid == resource.data.authorId;
      allow update, delete: if request.auth != null && 
        resource.data.authorId == request.auth.uid;
    }
    
    // Craft Rooms collection
    match /craftRooms/{roomId} {
      allow read: if resource == null || resource.data.isPublic == true || 
        (request.auth != null && resource.data.hostId == request.auth.uid);
      allow create: if request.auth != null && 
        request.auth.uid == resource.data.hostId;
      allow update, delete: if request.auth != null && 
        resource.data.hostId == request.auth.uid;
    }
    
    // Craft Room Participants collection
    match /craftRoomParticipants/{participantId} {
      allow read, write: if request.auth != null;
    }
    
    // Craft Room Messages collection
    match /craftRoomMessages/{messageId} {
      allow read, write: if request.auth != null;
    }
  }
}
